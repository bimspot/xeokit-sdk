    /**
 * @desc Default {@link ContextMenu} items for use with {@link Entity}s.
 *
 * @type {Object[]}
 */
const DefaultObjectContextMenuItems = [

    // First group of items
    [
        {
            title: "Fly to object",
            callback: function (context) {
                const entity = context.entity;
                const scene = entity.scene;
                const viewer = scene.viewer;
                viewer.cameraFlight.flyTo({
                    aabb: entity.aabb,
                    duration: 0.5
                }, () => {
                    setTimeout(function () {
                        // viewer.scene.xrayMaterial.edgeAlpha = 0.2;

                        scene.setObjectsHighlighted(scene.highlightedObjectIds, false);
                        // scene.setObjectsVisible(scene.xrayedObjectIds, false);
                    }, 500);
                });
            }
        },
        {
            title: "Show object in tree",
            callback: function (context) {

            }
        }
    ],

    // Second group of items
    [
        {
            title: "Hide object",
            callback: function (context) {
                context.entity.visible = false;
            }
        },
        {
            title: "X-Ray object",
            callback: function (context) {
                context.entity.xrayed = true;
            }
        },
        {
            title: "Select object",
            callback: function (context) {
                context.entity.selected = true;
            }
        },

        {
            title: "Hide other objects",
            callback: function (context) {
                const entity = context.entity;
                const scene = entity.scene;
                scene.setObjectsVisible(scene.visibleObjectIds, false);
                scene.setObjectsXRayed(scene.xrayedObjectIds, false);
                scene.setObjectsSelected(scene.selectedObjectIds, false);
                scene.setObjectsHighlighted(scene.highlightedObjectIds, false);
                entity.visible = true;

            }
        }
    ],

    // Third group of items
    [
        {
            title: "Show all objects",
            callback: function (context) {
                const scene = context.viewer.scene;
                scene.setObjectsVisible(scene.objectIds, true);
                scene.setObjectsXRayed(scene.xrayedObjectIds, false);
                scene.setObjectsSelected(scene.selectedObjectIds, false);
            }
        }
    ],
    // Fourth group of items
    [
        {
            title: "X-ray object",
            callback: function (context) {
                context.treeViewPlugin.withNodeTree(context.treeViewNode, (treeViewNode) => {
                    if (treeViewNode.objectId) {
                        const entity = context.viewer.scene.objects[treeViewNode.objectId];
                        if (entity) {
                            entity.xrayed = true;
                        }
                    }
                });
            }
        },
        {
            title: "X-ray other objects",
            callback: function (context) {
                const scene = context.viewer.scene;
                scene.setObjectsXRayed(scene.objectIds, true);
                scene.setObjectsSelected(scene.selectedObjectIds, false);
                scene.setObjectsHighlighted(scene.highlightedObjectIds, false);
                context.treeViewPlugin.withNodeTree(context.treeViewNode, (treeViewNode) => {
                    if (treeViewNode.objectId) {
                        const entity = scene.objects[treeViewNode.objectId];
                        if (entity) {
                            entity.xrayed = false;
                            entity.visible = true;
                        }
                    }
                });
            }
        },
        {
            title: "Clear X-ray",
            callback: function (context) {
                context.viewer.scene.setObjectsXRayed(context.viewer.scene.xrayedObjectIds, false);
            }
        }
    ]
];

export {DefaultObjectContextMenuItems};
